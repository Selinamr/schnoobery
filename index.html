function wKeydown(e) {

  if (wDone) return;

  if (e.key === 'Enter') handleKey('ENTER');

  else if (e.key === 'Backspace') handleKey('\u232B');

  else if (/^[a-zA-Z]$/.test(e.key)) handleKey(e.key.toUpperCase());

}


function handleKey(k) {

  if (wDone) return;

  const row = wGuesses.length;

  if (k === '\u232B') {

    if (wCurrent.length > 0) { wCurrent = wCurrent.slice(0, -1); updateRow(row); }

  } else if (k === 'ENTER') {

    if (wCurrent.length < 5) { document.getElementById('wmsg').textContent = '5 letters please.'; return; }

    showLoadingPopup(submitGuess);

  } else if (wCurrent.length < 5) {

    wCurrent += k;

    updateRow(row);

  }

}


function updateRow(row) {

  for (let c = 0; c < 5; c++) {

    const t = document.getElementById('t-' + row + '-' + c);

    t.textContent = wCurrent[c] || '';

    t.className = 'wtile' + (wCurrent[c] ? ' filled' : '');

  }

}


function submitGuess() {

  const row = wGuesses.length;

  const guess = wCurrent;

  const result = scoreWord(guess);

  wGuesses.push(guess);

  const keyState = {};

  for (let c = 0; c < 5; c++) {

    const t = document.getElementById('t-' + row + '-' + c);

    t.textContent = guess[c];

    t.className = 'wtile ' + result[c];

    const cur = keyState[guess[c]];

    if (!cur || cur === 'absent' || (cur === 'present' && result[c] === 'correct')) keyState[guess[c]] = result[c];

  }

  Object.entries(keyState).forEach(([k, s]) => {

    const el = document.getElementById('k-' + k);

    const cur = el && el.className.split(' ').find(c => ['correct','present','absent'].includes(c));

    if (el && (!cur || cur === 'absent' || (cur === 'present' && s === 'correct'))) el.className = 'wkey ' + s;

  });

  wCurrent = '';

  if (result.every(r => r === 'correct')) {

    wDone = true;

    document.removeEventListener('keydown', wKeydown);

    showWinPopup();

  } else if (wGuesses.length >= 6) {

    wDone = true;

    document.removeEventListener('keydown', wKeydown);

    showLosePopup();

  }

}


function scoreWord(guess) {

  const res = ['absent','absent','absent','absent','absent'];

  const pool = WORD.split('');

  for (let i = 0; i < 5; i++) { if (guess[i] === WORD[i]) { res[i] = 'correct'; pool[i] = null; } }

  for (let i = 0; i < 5; i++) {

    if (res[i] === 'correct') continue;

    const idx = pool.indexOf(guess[i]);

    if (idx > -1) { res[i] = 'present'; pool[idx] = null; }

  }

  return res;

}
